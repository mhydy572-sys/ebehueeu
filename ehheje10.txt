-- دالة مساعدة للبحث عن اللاعب بناءً على الحروف الأولى من اسمه
local function getPlayerByShortName(shortName)
    if shortName == "" or #shortName < 2 then return nil end
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and string.sub(string.lower(player.Name), 1, #shortName) == string.lower(shortName) then
            return player
        end
    end
    return nil
end

-- 5. الحلقة البرمجية المسؤولة عن النقل المستمر (Teleport Loop) وخيارات الملاحقة
RunService.RenderStepped:Connect(function()
    if not currentVictim or not currentVictim.Character then return end
    
    local myChar = LocalPlayer.Character
    local victimChar = currentVictim.Character
    
    local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
    local victimRoot = victimChar and victimChar:FindFirstChild("HumanoidRootPart")
    local victimHead = victimChar and victimChar:FindFirstChild("Head")
    
    -- نظام الكاميرا والمشاهدة التلقائية للضحية
    if isSpectateActive and victimChar:FindFirstChildOfClass("Humanoid") then
        workspace.CurrentCamera.CameraSubject = victimChar:FindFirstChildOfClass("Humanoid")
    end
    
    -- إلغاء قوى الجاذبية والحركة لمنع اللعبة من دفع اللاعب بعيداً أثناء الالتصاق
    if myRoot and (isBangBackActive or isBangFaceActive or isHeadSitActive or isBagBackActive) then
        myRoot.Velocity = Vector3.new(0, 0, 0)
        
        -- التموضع المستمر خلف الضحية أو أمامها بناءً على الزر المفعّل
        if isBangBackActive and victimRoot then
            myRoot.CFrame = victimRoot.CFrame * CFrame.new(0, 0, 1.1)
        elseif isBangFaceActive and victimRoot then
            myRoot.CFrame = victimRoot.CFrame * CFrame.new(0, 0, -1.1) * CFrame.Angles(0, math.pi, 0)
        elseif isHeadSitActive and victimHead then
            myRoot.CFrame = victimHead.CFrame * CFrame.new(0, 1.2, 0)
        elseif isBagBackActive and victimRoot then
            myRoot.CFrame = victimRoot.CFrame * CFrame.new(0, 0.5, 0.8)
        end
    end
end)

    end

-- تشغيل الـ Loop المشترك لتوليد ألوان الـ RGB (الألوان السبعة) وتطبيقها بالتناوب
task.spawn(function()
    while task.wait() do
        if hairRGB or nameRGB or bioRGB or bodyRGB then
            -- حساب قيمة اللون المتحرك بناءً على الوقت والسرعة المحددة
            local hue = (tick() * (rgbSpeed / 10)) % 1
            local colorRGB = Color3.fromHSV(hue, 1, 1)
            
            -- 1. تغيير لون الاسم RGB
            if nameRGB and nameRemote then
                local colorRemote = RE:FindFirstChild("1RPNam1eColo1r")
                if colorRemote then
                    colorRemote:FireServer("PickingRPNameColor", colorRGB)
                end
            end
            
            -- 2. تغيير لون البايو RGB
            if bioRGB and nameRemote then
                local colorRemote = RE:FindFirstChild("1RPNam1eColo1r")
                if colorRemote then
                    colorRemote:FireServer("PickingRPBioColor", colorRGB)
                end
            end
            
            -- 3. تغيير لون شعر الشخصية
            if hairRGB then
                local char = Players.LocalPlayer.Character
                if char then
                    for _, accessory in ipairs(char:GetChildren()) do
                        if accessory:IsA("Accessory") and accessory:FindFirstChild("Handle") then
                            -- فحص إذا كان الإكسسوار عبارة عن شعر لتغيير لونه
                            if accessory.Handle:FindFirstChildOfClass("SpecialMesh") then
                                accessory.Handle.Color = colorRGB
                            end
                        end
                    end
                end
            end
            
            -- 4. تغيير لون الجسم RGB
            if bodyRGB then
                local char = Players.LocalPlayer.Character
                if char then
                    for _, part in ipairs(char:GetChildren()) do
                        if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                            part.Color = colorRGB
                        end
                    end
                end
            end
            
            task.wait(0.1) -- متسع بسيط لتفادي الـ Lag الناتج عن إرسال ريموتات بكثافة
        end
    end
end)

        if effect:IsA("BloomEffect") or effect:IsA("BlurEffect") or effect:IsA("ColorCorrectionEffect") or effect:IsA("SunRaysEffect") or effect:IsA("Atmosphere") then

        end
    end
end

-- دالة لمراقبة أحداث لاعب محدد (الموت والشات)
local function monitorPlayer(player)
    if player == LocalPlayer then return end

    -- مراقبة الشات (Chatted)
    player.Chatted:Connect(function(message)
        if chatNotify then
            sendAlert("💬 شات جديد", player.DisplayName .. ": " .. message)
        end
    end)

    -- مراقبة الموت (Character & Humanoid Death)
    player.CharacterAdded:Connect(function(character)
        local humanoid = character:WaitForChild("Humanoid", 5)
        if humanoid then
            humanoid.Died:Connect(function()
                if deathNotify then
                    sendAlert("💀 حالة وفاة", "المستهدف [" .. player.DisplayName .. "] تم دعسه!")
                end
            end)
        end
    end)
    
    -- فحص إذا كان الشخص الذي دخل يمتلك رتبة أدمن أو مطور في روبلوكس
    if adminNotify then
        if knownAdmins[player.Name] or player:GetRankInGroup(1200769) >= 200 or player.AccountAge < 1 then 
            sendAlert("⚠️ تحذير أمني", "انتبه! الأدمن أو الحساب المشبوه [" .. player.Name .. "] دخل السيرفر الآن!")
        end
    end
end

-- تشغيل المراقبة على اللاعبين الموجودين حالياً بالسيرفر
for _, player in ipairs(Players:GetPlayers()) do
    monitorPlayer(player)
end

    if joinNotify then
        sendAlert("📥 دخول لاعب", player.DisplayName .. " (@" .. player.Name .. ") دخل السيرفر!")
    end
    monitorPlayer(player)
end)

    end
end)

local function playTargetSound(soundId, speed, volume, maxDistance, loop)
    -- البحث عن الصوت القديم وحذفه لتجنب التداخل
    local oldSound = workspace:FindFirstChild("NeutronExploitSound")
    if oldSound and not loop then oldSound:Destroy() end
    
    local sound = oldSound
    if not sound then
        sound = Instance.new("Sound")
        sound.Name = "NeutronExploitSound"
        sound.Parent = workspace:FindFirstChild("HumanoidRootPart", true) or workspace
    end
    
    sound.SoundId = "rbxassetid://" .. tostring(soundId)
    sound.PlaybackSpeed = speed
    sound.Volume = volume
    sound.MaxDistance = maxDistance
    sound.RollOffMaxDistance = maxDistance
    
    if not sound.IsPlaying then
        sound:Play()
    end
    
    if not loop then
        sound.Ended:Connect(function()
            sound:Destroy()
        end)
    end
    return sound
end

    if orbitLoop then orbitLoop:Disconnect() orbitLoop = nil end
    if orbitBallInstance then orbitBallInstance:Destroy() orbitBallInstance = nil end

local function orbitBall(targetChar, modeName)
    if not targetChar or not targetChar:FindFirstChild("HumanoidRootPart") then return end
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local backpack = LocalPlayer:WaitForChild("Backpack")
    local serverBalls = Workspace:WaitForChild("WorkspaceCom"):WaitForChild("001_SoccerBalls")
    
    local toolRemote = ReplicatedStorage:FindFirstChild("RE") and ReplicatedStorage.RE:FindFirstChild("1Too1l")
    if not backpack:FindFirstChild("SoccerBall") and not char:FindFirstChild("SoccerBall") and toolRemote then
        toolRemote:InvokeServer("PickingTools", "SoccerBall")
    end
    
    repeat task.wait() until backpack:FindFirstChild("SoccerBall") or char:FindFirstChild("SoccerBall")
    local ballTool = backpack:FindFirstChild("SoccerBall")
    if ballTool then ballTool.Parent = char end
    
    repeat task.wait() until serverBalls:FindFirstChild("Soccer" .. LocalPlayer.Name)
    local orb = serverBalls:FindFirstChild("Soccer" .. LocalPlayer.Name)
    if not orb then return end
    
    orb.Name = modeName
    orb.CanCollide = false
    orb.Massless = true
    orb.CustomPhysicalProperties = PhysicalProperties.new(0.0001, 0, 0)
    
    if orb:FindFirstChild("FlingPower") then orb.FlingPower:Destroy() end
    if OrbitSettings.Fling then
        local bv = Instance.new("BodyVelocity")
        bv.Name = "FlingPower"
        bv.Velocity = Vector3.new(9e8, 9e8, 9e8)
        bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
        bv.P = 9e900
        bv.Parent = orb
    end
    
    local loop = Instance.new("BindableEvent", char)
    loop.Name = modeName .. "Loop"
    
    task.spawn(function()
        while loop.Parent and orb and orb.Parent and targetChar and targetChar.Parent do
            local angle = tick() * OrbitSettings.Speed
            local offset = CFrame.new(math.cos(angle) * OrbitSettings.Distance, OrbitSettings.Height, math.sin(angle) * OrbitSettings.Distance)
            orb.CFrame = targetChar.HumanoidRootPart.CFrame * offset
            task.wait(1/60)
        end
    end)
end

local isProcessing = false

-- وظيفة جلب وحجز اللاعب بالأريكة (Couch Bring)
local function ExecuteCouchBringClear()
    if isProcessing then return end
    if not getgenv().Target then return end
    
    local targetPlayer = Players:FindFirstChild(getgenv().Target)
    local char = LocalPlayer.Character
    if not targetPlayer or not char then return end

    local hum = char:FindFirstChildOfClass("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    local targetRoot = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    local targetHum = targetPlayer.Character and targetPlayer.Character:FindFirstChildOfClass("Humanoid")

    if not hum or not root or not targetRoot or not targetHum then return end

    isProcessing = true
    local originalCFrame = root.CFrame
    local remotes = ReplicatedStorage:WaitForChild("RE", 5)
    if not remotes then isProcessing = false return end
    
    pcall(function()
        remotes:WaitForChild("1Clea1rTool1s"):FireServer("ClearAllTools")
        task.wait(0.15)
        remotes:WaitForChild("1Too1l"):InvokeServer("PickingTools", "Couch")
    end)
    
    local couch = LocalPlayer.Backpack:WaitForChild("Couch", 4)
    if not couch then 
        isProcessing = false
        return 
    end
    couch.Parent = char
    
    task.wait(0.15)
    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.F, false, game)
    hum:SetStateEnabled(Enum.HumanoidStateType.Seated, false)

    local bp = Instance.new("BodyPosition")
    bp.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bp.D = 100
    bp.P = 10000
    bp.Parent = targetRoot

    local startTime = tick()
    while tick() - startTime < 6 do
        if not targetHum or targetHum.Sit or not flingActive then break end
        
        local rot = CFrame.Angles(math.rad(math.random(-90, 90)), math.rad(math.random(-90, 90)), math.rad(math.random(-90, 90)))
        local offset = Vector3.new(math.random(-3, 3), 2, math.random(-3, 3))
        
        root.CFrame = CFrame.new(targetRoot.Position + offset) * rot
        bp.Position = root.Position
        task.wait(0.05)
    end

    bp:Destroy()
    root.Velocity = Vector3.zero
    root.RotVelocity = Vector3.zero
    root.CFrame = originalCFrame
    
    task.wait(0.5)
    
    pcall(function() remotes:WaitForChild("1Clea1rTool1s"):FireServer("ClearAllTools") end)
    local checkTool = char:FindFirstChild("Couch") or LocalPlayer.Backpack:FindFirstChild("Couch")
    if checkTool then checkTool:Destroy() end

    hum:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    hum.WalkSpeed = 0
    hum.JumpPower = 0
    
    local stopTime = tick()
    while tick() - stopTime < 2 do
        root.Velocity = Vector3.zero
        root.RotVelocity = Vector3.zero
        root.CFrame = originalCFrame
        task.wait()
    end
    
    hum.WalkSpeed = 16
    hum.JumpPower = 50
    isProcessing = false
end

-- ميزة طرد وقذف الأريكة الخارق (Couch Fling)
-- ------------------------------------------
local flingActive = false
local flingSpeed = 1

CouchTab:AddSlider({
    Title = "سرعة رمي وسرعة الأريكة",
    Min = 0.1,
    Max = 7,
    Default = 1,
    Increment = 0.1,
    Callback = function(value)
        flingSpeed = value
    end
})

CouchTab:AddToggle({
    Title = "تشغيل قذف الأريكة (اول مستوا)",
    Default = false,
    Callback = function(state)
        flingActive = state
        if state then
            task.spawn(function()
                while flingActive do
                    local target = Players:FindFirstChild(getgenv().Target or "")
                    if target and target.Character then
                        local char = LocalPlayer.Character
                        local root = char and char:FindFirstChild("HumanoidRootPart")
                        local tRoot = target.Character:FindFirstChild("HumanoidRootPart")
                        local tHum = target.Character:FindFirstChildOfClass("Humanoid")
                        local hum = char and char:FindFirstChildOfClass("Humanoid")
                        local remotes = ReplicatedStorage:FindFirstChild("RE")

                        if root and tRoot and tHum and hum and remotes then
                            pcall(function()
                                remotes:FindFirstChild("1Clea1rTool1s"):FireServer("ClearAllTools")
                                task.wait(0.2)
                                remotes:FindFirstChild("1Too1l"):InvokeServer("PickingTools", "Couch")
                            end)
                            
                            local original = root.CFrame
                            local tool = LocalPlayer.Backpack:FindFirstChildOfClass("Tool")
                            if tool then tool.Parent = char end
                            
                            Workspace.FallenPartsDestroyHeight = -math.huge
                            
                            local bv = Instance.new("BodyVelocity")
                            bv.Name = "CouchFlingForce"
                            bv.Velocity = Vector3.new(9e8, 9e8, 9e8)
                            bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                            bv.Parent = root
                            
                            hum:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
                            hum.PlatformStand = false
                            Camera.CameraSubject = tRoot
                            
                            local angle = 0
                            while flingActive and target and target.Character and target.Character:FindFirstChildOfClass("Humanoid") do
                                tHum = target.Character:FindFirstChildOfClass("Humanoid")
                                tRoot = target.Character:FindFirstChild("HumanoidRootPart")
                                if not tRoot or tHum.Sit or tRoot.Position.Y >= 70000 then break end
                                
                                local moveDir = tHum.MoveDirection
                                local offset = Vector3.new(0, 0, 0)
                                if moveDir.Magnitude >= 0.01 then
                                    local speed = tHum.WalkSpeed
                                    local studs = 13
                                    if speed > 16 then studs = studs + math.floor((speed - 16) / 9) * 9 end
                                    offset = moveDir.Unit * studs
                                end
                                
                                angle = angle + (30 * flingSpeed)
                                root.CFrame = CFrame.new(tRoot.Position + offset + Vector3.new(0, 1, 0)) * CFrame.Angles(math.rad(angle), 0, 0)
                                root.Velocity = Vector3.new(9e8, 9e8, 9e8)
                                root.RotVelocity = Vector3.new(9e8, 9e8, 9e8)
                                task.wait(0.4 / flingSpeed)
                            end
                            
                            bv:Destroy()
                            hum:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
                            hum.PlatformStand = false
                            root.CFrame = original
                            Camera.CameraSubject = hum
                            
                            for _, p in pairs(char:GetDescendants()) do 
                                if p:IsA("BasePart") then 
                                    p.Velocity = Vector3.zero 
                                    p.RotVelocity = Vector3.zero 
                                end 
                            end
                            hum:UnequipTools()
                        end
                    end
                    task.wait(0.4 / flingSpeed)
                end
            end)
        else
            -- إرجاع وضع الكاميرا الطبيعي فور الإيقاف وايقاف الدوران
            local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            if hum then Camera.CameraSubject = hum end
        end
    end
})

local function findNewJet(usedJets, storage)
    if not storage then return nil end
    for _, jet in ipairs(storage:GetChildren()) do
        if jet:IsA("Model") and jet:FindFirstChild("NoUse") and not table.find(usedJets, jet) then
            return jet
        end
    end
    return nil
end

local function startJetSpam()
    local humanoidRootPart = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not humanoidRootPart then return end
    local detector, storage = getJetComponents()
    if not detector or not storage then return end

    JetSystem.originalPosition = humanoidRootPart.CFrame
    humanoidRootPart.CFrame = JetSystem.teleportPosition
    JetSystem.spawnActive = true

    task.spawn(function()
        local destinationIndex = 1
        local usedJets = {}
        while JetSystem.spawnActive do
            if detector then fireclickdetector(detector) end
            task.wait(Config.jetSpawnDelay)
            local newJet = findNewJet(usedJets, storage)
            if newJet then
                local height = (#usedJets + 1) * 6
                local baseDestination = JetSystem.destinations[destinationIndex] or JetSystem.destinations[1]
                local target = baseDestination + Vector3.new(0, height, 0)
                local primaryPart = newJet:FindFirstChild("NoUse") or newJet:FindFirstChildWhichIsA("BasePart")
                if primaryPart then
                    newJet.PrimaryPart = primaryPart
                    pcall(function() newJet:SetPrimaryPartCFrame(target) end)
                    table.insert(usedJets, newJet)
                end
            end
            destinationIndex = (destinationIndex % #JetSystem.destinations) + 1
            task.wait(0.5)
        end
    end)
end

local function stopJetSpam()
    JetSystem.spawnActive = false
    if JetSystem.originalPosition then
        local humanoidRootPart = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if humanoidRootPart then humanoidRootPart.CFrame = JetSystem.originalPosition end
    end
end

local function startTornado()
    ShipSystem.currentShip = spawnPirateShip()
    if not ShipSystem.currentShip then return end
    local seat = ShipSystem.currentShip:FindFirstChild("Body") and ShipSystem.currentShip.Body:FindFirstChild("VehicleSeat")
    if not seat then return end
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("Humanoid") or not character:FindFirstChild("HumanoidRootPart") then return end
    
    repeat task.wait(0.1) character.HumanoidRootPart.CFrame = seat.CFrame * CFrame.new(0, 1, 0) until character.Humanoid.SeatPart == seat
    task.delay(4, function() if character.Humanoid.SeatPart then character.Humanoid.Sit = false end character.HumanoidRootPart.CFrame = CFrame.new(0, 0, 0) end)
    ShipSystem.tornadoActive = true
    
    task.spawn(function()
        local re = ReplicatedStorage:WaitForChild("RE", 5) and ReplicatedStorage.RE:FindFirstChild("1Player1sCa1r")
        while ShipSystem.tornadoActive and ShipSystem.currentShip and ShipSystem.currentShip.Parent and re do 
            pcall(function() re:FireServer("Flip") end) 
            task.wait(0.5) 
        end
    end)
    
    local waypoints = {Vector3.new(-16, 0, -47), Vector3.new(-110, 0, -45), Vector3.new(16, 0, -55)}
    local currentIndex, nextIndex, moveSpeed, rotationSpeed, progress, currentRotation = 1, 2, 30, math.rad(720), 0, 0
    local connection
    connection = RunService.Heartbeat:Connect(function(dt)
        if not ShipSystem.tornadoActive or not ShipSystem.currentShip or not ShipSystem.currentShip.PrimaryPart then connection:Disconnect() return end
        local startPos, endPos = waypoints[currentIndex], waypoints[nextIndex]
        progress = progress + (moveSpeed * dt) / (startPos - endPos).Magnitude
        if progress >= 1 then progress = 0 currentIndex = nextIndex nextIndex = (nextIndex % #waypoints) + 1 end
        currentRotation = currentRotation + rotationSpeed * dt
        pcall(function() ShipSystem.currentShip:SetPrimaryPartCFrame(CFrame.new(startPos:lerp(endPos, progress)) * CFrame.Angles(0, currentRotation, 0)) end)
    end)
end

        task.wait(0.5)
        local detector = workspace:WaitForChild("WorkspaceCom"):WaitForChild("001_DayCare").Tools:WaitForChild("FireX"):WaitForChild("ClickDetector")

        repeat
            if detector then fireclickdetector(detector) end
            task.wait()
        until #LocalPlayer.Backpack:GetChildren() >= totalToolsNeeded

        local tools = LocalPlayer.Backpack:GetChildren()
        local gripMap, startPos, pixelSize, toolIndex, offsetX = {}, Vector3.new(0, -5, 0), Vector3.new(1.2, 1.2, 0), 1, 0

        for i = 1, #text do
            local char = text:sub(i, i):upper()
            local map = CHAR_MAP[char]
            if map then
                for y = 1, #map do
                    for x = 1, #map[y] do
                        if map[y]:sub(x, x) == "1" then
                            local t = tools[toolIndex]
                            if t then gripMap[t] = CFrame.new(startPos.X + (offsetX + x - 1) * pixelSize.X, startPos.Y + (y - 1) * pixelSize.Y, startPos.Z) toolIndex = toolIndex + 1 end
                        end
                    end
                end
                offsetX = offsetX + #map[1] + 1
            end
        end

        hrp.CFrame = CFrame.new(oldPos)
        task.wait(2)
        for tool, grip in pairs(gripMap) do tool.Grip = grip end
        task.wait(1)
        for i = 1, totalToolsNeeded do if tools[i] then tools[i].Parent = character end end
        getgenv()._Running = false
    end
